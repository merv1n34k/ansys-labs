\section{Теоретичні відомості}
% ═══════════════════════════════════════════════════════════════════

\subsection{Механізми теплообміну}

\textbf{Теплопровідність (Conduction)} --- передача тепла через тверді тіла або нерухому рідину. Закон Фур'є:
\begin{equation}
  q = -\lambda\,\frac{\partial T}{\partial x},
\end{equation}
де $q$~--- тепловий потік (Вт/м$^2$), $\lambda$~--- коефіцієнт теплопровідності (Вт/(м$\cdot$К)).

\textbf{Конвекція (Convection)} --- передача тепла рухомою рідиною. Закон Ньютона--Ріхмана:
\begin{equation}
  q = h\,(T_{\mathrm{wall}} - T_{\mathrm{fluid}}),
\end{equation}
де $h$~--- коефіцієнт тепловіддачі (Вт/(м$^2\cdot$К)).

\textbf{Випромінювання (Radiation)} --- передача тепла електромагнітними хвилями. Закон Стефана--Больцмана:
\begin{equation}
  q = \varepsilon\,\sigma\,T^4,
  \quad \sigma = 5{,}67 \times 10^{-8}~\text{Вт/(м$^2\cdot$К$^4$)}.
\end{equation}

\subsection{Рівняння енергії}

Для рідини з конвекцією:
\begin{equation}
  \rho\, c_p\left(\frac{\partial T}{\partial t}
  + \mathbf{U}\cdot\nabla T\right)
  = \nabla\cdot(\lambda\,\nabla T) + S_E.
\end{equation}

Для твердого тіла (без конвекції):
\begin{equation}
  \rho\, c_p\,\frac{\partial T}{\partial t}
  = \nabla\cdot(\lambda\,\nabla T) + S_E.
\end{equation}

\subsection{Безрозмірні критерії}

Число Нуссельта (Nusselt number):
\begin{equation}
  \mathrm{Nu} = \frac{h\,D}{\lambda}.
\end{equation}

Число Прандтля (Prandtl number):
\begin{equation}
  \mathrm{Pr} = \frac{\mu\, c_p}{\lambda} = \frac{\nu}{a}.
\end{equation}

\subsection{Conjugate Heat Transfer (CHT)}

Спряжений теплообмін передбачає одночасне моделювання теплообміну в рідині (конвекція + теплопровідність) та в твердому тілі (теплопровідність). На інтерфейсі Fluid--Solid забезпечується безперервність:
\begin{equation}
  T_{\mathrm{solid}}\big|_{\mathrm{int}} = T_{\mathrm{fluid}}\big|_{\mathrm{int}},
  \qquad
  q_{\mathrm{solid}}\big|_{\mathrm{int}} = q_{\mathrm{fluid}}\big|_{\mathrm{int}}.
\end{equation}

\section{Хронологічна послідовність виконання}
% ═══════════════════════════════════════════════════════════════════

\begin{enumerate}
  \item \textbf{Підготовка геометрії.}
    Використано модель ємності з Практичної роботи~№1. Створено внутрішній об'єм ємності як рідинний домен. Інтегровано трубчастий контур (змійовик / пряма труба) з входом та виходом. Геометрія складається з трьох окремих тіл:
    \begin{itemize}
      \item Внутрішній об'єм ємності (Fluid --- Water);
      \item Тіло контуру (Solid --- Copper);
      \item Внутрішній об'єм контуру (Fluid --- Water).
    \end{itemize}

  \item \textbf{Побудова сітки (Meshing).}
    Побудовано об'ємну сітку з ущільненням у зонах контактів (Fluid--Solid interfaces).

  \item \textbf{Налаштування доменів (CFX-Pre).}
    \begin{itemize}
      \item Fluid Domain (ємність): Water, $T = 12$~°C, Inlet $V = 0{,}006$~м/с;
      \item Solid Domain (контур): Copper;
      \item Fluid Domain (контур): Water, $T = 108$~°C, Inlet $V = 1{,}4$~м/с;
      \item Reference Pressure: 1~атм;
      \item Turbulence model: $k$-$\varepsilon$;
      \item Heat Transfer model: Total Energy.
    \end{itemize}

  \item \textbf{Інтерфейси.}
    Створено Fluid--Solid interfaces між рідинними та твердим доменами.

  \item \textbf{Розрахунок та аналіз.}
    Запущено стаціонарний розрахунок. Перевірено збіжність.
\end{enumerate}
